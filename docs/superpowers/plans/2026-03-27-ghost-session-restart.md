**Goal:** Add a right-click context menu on ghost session rows with "Dismiss" and "Open in VS Code" options, where "Open in VS Code" writes a temporary tasks.json for auto-launching Claude and cleans it up when the session registers.

**Architecture:** Ghost rows get per-row right-click handlers that dispatch to either the ghost context menu (Dismiss/Open in VS Code) or the existing visibility checklist. The tasks.json lifecycle is managed via hash tracking in the existing session state file — write on menu click, cleanup on session discovery.

**Tech Stack:** Python 3.11+, Tkinter, hashlib (sha256), subprocess, pathlib

```python
import hashlib


def write_vscode_tasks_json(*, cwd: str) -> str | None:
    """Write the VS Code tasks.json template to <cwd>/.vscode/tasks.json.

    Returns the hash if written or if existing file matches template.
    Returns None if existing file has different content or CWD doesn't exist.
    """
    cwd_path = Path(cwd)
    if not cwd_path.is_dir():
        logger.warning("cwd does not exist, skipping tasks.json write cwd=%s", cwd)
        return None

    tasks_path = cwd_path / ".vscode" / "tasks.json"
    if tasks_path.exists():
        existing_hash = hashlib.sha256(
            tasks_path.read_bytes()
        ).hexdigest()
        if existing_hash == config.VSCODE_TASKS_JSON_HASH:
            logger.debug("tasks.json already matches template cwd=%s", cwd)
            return config.VSCODE_TASKS_JSON_HASH
        logger.info(
            "tasks.json exists with different content, skipping write cwd=%s", cwd
        )
        return None

    tasks_path.parent.mkdir(parents=True, exist_ok=True)
    tasks_path.write_text(config.VSCODE_TASKS_JSON_TEMPLATE, encoding="utf-8")
    logger.info("wrote tasks.json cwd=%s", cwd)
    return config.VSCODE_TASKS_JSON_HASH


def cleanup_vscode_tasks_json(*, cwd: str, expected_hash: str) -> bool:
    """Delete <cwd>/.vscode/tasks.json if its hash matches expected_hash.

    Returns True if file was deleted, False otherwise.
    Also removes .vscode/ directory if empty after deletion.
    """
    tasks_path = Path(cwd) / ".vscode" / "tasks.json"
    if not tasks_path.exists():
        return False

    current_hash = hashlib.sha256(tasks_path.read_bytes()).hexdigest()
    if current_hash != expected_hash:
        logger.info(
            "tasks.json hash mismatch, keeping file cwd=%s", cwd
        )
        return False

    tasks_path.unlink()
    logger.info("cleaned up tasks.json cwd=%s", cwd)

    # Remove .vscode/ if empty
    vscode_dir = tasks_path.parent
    try:
        vscode_dir.rmdir()  # only succeeds if empty
        logger.debug("removed empty .vscode/ dir cwd=%s", cwd)
    except OSError:
        pass  # directory has other files, leave it

    return True
```

    def _open_in_vscode(self, session: SessionInfo):
        """Write tasks.json and launch VS Code for a ghost session."""
        result_hash = write_vscode_tasks_json(cwd=session.cwd)
        if result_hash:
